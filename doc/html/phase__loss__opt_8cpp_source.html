55 void Phase::avg_fitness(double *soln, const int K, double *fitarray) {
56  const double loss = 0.0;//loss level
57  dcmplx sharp(0.0, 0.0);
58  bool dect_result;
59  double PHI, phi, coin, PHI_in;
60  int m, k, d;
61  double error = 0.0;
62 
63  WK_state();
64  for(k = 0; k < K; ++k) {
65  phi = rng->next_urand() * (upper - lower) + lower;
66  PHI = 0;
67  //copy input state: the optimal solution across all compilers is memcpy:
68  //nadeausoftware.com/articles/2012/05/c_c_tip_how_copy_memory_quickly
69  memcpy(state, input_state, (num + 1)*sizeof(dcmplx));
70  //measurement
71  d = 0;
72  for (m = 0; m < num; ++m) {
73  // This loop is the most critical part of the entire program. It
74  // executes K*num=10*num^3 times on each call of avg_fitness. All
75  // optimization should focus on this loop.
76 
77  //randomly decide whether loss occurs
78  coin = rng->next_urand();
79 
80  if(coin <= loss) {
81  state_loss(num - m); //update only the state using loss function
82  }
83  else {
84  PHI_in = rng->next_grand(PHI, THETA_DEV);
85  PHI_in = mod_2PI(PHI_in);//noisy PHI
86  dect_result = noise_outcome(phi, PHI_in, num - m);
87  if (dect_result == 0) {
88  PHI = PHI - soln[d++];
89  }
90  else {
91  PHI = PHI + soln[d++];
92  }
93  PHI = mod_2PI(PHI);
94  }
95  }
96  sharp.real() = sharp.real() + cos(phi - PHI);
97  sharp.imag() = sharp.imag() + sin(phi - PHI);
98  error += abs(phi - PHI);
99  }
100  fitarray[0] = abs(sharp) / double(K);
101  fitarray[1] = error / double(K);
102 
103  }

123

125  double n_part[num + 1];

129

132 
133  /*factors in d_matrix calculated*/

135  for (n = 0; n <= num; ++n) {
136  n_part[n] = pow(-1.0, n) * sqrtfac_mat[n] * sqrtfac_mat[num - n] * cosN * pow(tan_beta, n);
137  }
138  for (k = 0; k <= num; ++k) {
139  k_part[k] = 1 / pow(-1.0, k) * sqrtfac_mat[k] * sqrtfac_mat[num - k] * pow(1 / tan_beta, k);
140  }
141 
142  for (n = 0; n <= num; ++n) { //we have N+1 state b/c we include n=0 and n=N.
143  temp = 0;
144  input_state[n].real() = 0;
145  input_state[n].imag() = 0;
146  for (k = 0; k <= num; ++k) {
147  s_part = cal_spart(n, k, num);
148  temp = s_part * k_part[k] * sin((k + 1) * M_PI / (num + 2));
149  input_state[n].real() = input_state[n].real() + temp * cos(M_PI / 2.0 * (k - n));
150  input_state[n].imag() = input_state[n].imag() + temp * sin(M_PI / 2.0 * (k - n));
151  }//end k
152  // FIXED:This correction corrects the probability overshoot.
153  input_state[n].real() = input_state[n].real() * n_part[n] / sqrt(num / 2.0 + 1);
154  input_state[n].imag() = input_state[n].imag() * n_part[n] / sqrt(num / 2.0 + 1);
155  }//end n

159  //calculating the Wigner d-matrix element
160  int s;
161  int s_min;
162  int s_max;
163  double s_part = 0;
164 
165  //find lower limit
166  if(n - k >= 0) {
167  s_min = 0;
168  }
169  else if(n - k < 0) {
170  s_min = k - n;
171  }
172  else {}
173  //find upper limit
174  if(k <= N - n) {
175  s_max = k;
176  }
177  else if(k > N - n) {
178  s_max = N - n;
179  }
180  else {}
181 
182  //calculating the s_part
183  for (s = s_min; s <= s_max; ++s) {
184  s_part = s_part + pow(-1.0, s) * overfac_mat[k - s] * overfac_mat[s] * overfac_mat[n - k + s] * overfac_mat[N - n - s] * pow(tan_beta, 2 * s);
185  }

254 inline bool Phase::noise_outcome(const double phi, const double PHI, const int N) {
255  //N is the number of photons currently available, not equal to 'num'
256  const double theta = (phi - PHI) / 2.0;
257  const double cos_theta = cos(theta) / sqrt_cache[N];
258  const double sin_theta = sin(theta) / sqrt_cache[N];
259  //noise in operator: currently not in use
260  const double oper_n0 = rng->next_grand(0.0, DEV_N);//n_x
261  const double oper_n2 = rng->next_grand(0.0, DEV_N);//n_z
262  const double oper_n1 = sqrt(1.0 - (oper_n0 * oper_n0 + oper_n2 * oper_n2));
263  const dcmplx U00(sin_theta * oper_n1, -oper_n0 * sin_theta);
264  const dcmplx U01(cos_theta, oper_n2 * sin_theta);
265  const dcmplx U10(cos_theta, -oper_n2 * sin_theta);
266  const dcmplx U11(sin_theta * oper_n1, sin_theta * oper_n0);
267  int n;
268  double prob = 0.0;
269  for (n = 0; n < N; ++n) {
270  //if C_0 is measured
271  update0[n] = state[n + 1] * U00 * sqrt_cache[n + 1] + state[n] * U01 * sqrt_cache[N - n];
272  prob += abs(update0[n] * conj(update0[n]));
273  }
274 
275  if (rng->next_urand() <= prob) {
276  //measurement outcome is 0
277  state[N] = 0;
278  prob = 1.0 / sqrt(prob);
279  for(n = 0; n < N; ++n) {
280  state[n] = update0[n] * prob;
281  }
282  return 0;
283  }
284  else {
285  //measurement outcome is 1
286  prob = 0;
287  for(n = 0; n < N; ++n) {
288  state[n] = state[n + 1] * U10 * sqrt_cache[n + 1] - state[n] * U11 * sqrt_cache[N - n];

290  }
291  state[N] = 0;
292  prob = 1.0 / sqrt(prob);

295  }

300 inline void Phase::state_loss(const int N) {
301  //state update if the photon is loss
302  double total = 0;
303  double factor = 1 / sqrt(2 * N);
304  for(int n = 0; n < N; ++n) {
305  state[n] = (state[n] * sqrt_cache[N - n] + state[n + 1] * sqrt_cache[n + 1]) * factor;
306  total += state[n].real() * state[n].real() + state[n].imag() * state[n].imag();
307  }
308  state[N] = 0;
309  //Necessary state renormalization
310  for(int n = 0; n < N; ++n) {
311  state[n] = state[n] / sqrt(total);
