347 void OptAlg::linear_fit(int data_size, double *x, double *y, double *slope, double *intercept, double *mean_x) {
348  if(data_size <= 0) {
349  throw invalid_argument("data_size must be positive.");
350  }
351  double sum_x = 0;
352  double sum_xx = 0;
353  double sum_y = 0;
354  double sum_xy = 0;
355 
356  for(int v = 0; v < data_size; ++v) {
357  sum_x = sum_x + x[v];
358  sum_xx = sum_xx + x[v] * x[v];
359  sum_y = sum_y + y[v];
360  sum_xy = sum_xy + x[v] * y[v];
361  *mean_x = *mean_x + x[v];
362  }
363 
364  *mean_x = *mean_x / double(data_size);
365  *slope = (sum_xy - sum_y * sum_x / double(data_size)) / (sum_xx - sum_x * sum_x / double(data_size));
366  *intercept = sum_y / double(data_size) - *slope * sum_x / double(data_size);
367  }
